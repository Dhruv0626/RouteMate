const mongoose = require("mongoose");
const { Schema } = mongoose;

const UserSchema = new Schema(
  {
    // ── Core Identity ──────────────────────────────────────────────────────
    name:         { type: String, required: true, trim: true },
    email:        { type: String, required: true, unique: true, lowercase: true },
    mobile_no:    { type: String, required: true, unique: true },
    password:     { type: String, required: true },            // bcrypt hashed
    role:         { type: String, enum: ["passenger", "driver", "admin"], default: "passenger" },
    profileImage: { type: String, default: "" },               // Cloudinary URL

    // ── Account Status ─────────────────────────────────────────────────────
    isVerified:   { type: Boolean, default: false },
    isBlocked:    { type: Boolean, default: false },
    fcmToken:     { type: String, default: "" },               // push notification token

    // ── Passenger Stats (only used when role = passenger) ──────────────────
    passengerStats: {
      totalTrips:             { type: Number, default: 0 },
      totalDistanceTraveled:  { type: Number, default: 0 },    // in km
      averageRating:          { type: Number, default: 5.0 },  // rated by drivers
      totalRatingsReceived:   { type: Number, default: 0 },
      trustScore:             { type: Number, default: 100 },  // 0–100
    },
  },
  { timestamps: true }
);

UserSchema.index({ email: 1 }, { unique: true });
UserSchema.index({ mobile_no: 1 }, { unique: true });

module.exports = mongoose.model("User", UserSchema);

const DriverProfileSchema = new Schema(
  {
    user: { type: Schema.Types.ObjectId, ref: "User", required: true, unique: true },
    bio:  { type: String, default: "" },

    // ── License ────────────────────────────────────────────────────────────
    // image = single collage: LEFT half = front side, RIGHT half = back side
    license: {
      number: { type: String, default: "" },
      expiry: { type: Date },
      image:  { type: String, default: "" },                   // Cloudinary URL (collage)
    },

    // ── Aadhar ─────────────────────────────────────────────────────────────
    // image = single collage: LEFT half = front side, RIGHT half = back side
    aadhar: {
      number: { type: String, default: "" },
      image:  { type: String, default: "" },                   // Cloudinary URL (collage)
    },

    // ── Vehicle ────────────────────────────────────────────────────────────
    vehicle: {
      name:             { type: String, default: "" },         // e.g. "Swift Dzire"
      type: {
        type: String,
        enum: ["Sedan", "SUV", "Hatchback", "Auto", "Bike"],
        default: "Sedan",
      },
      number:           { type: String, default: "" },         // Registration number
      color:            { type: String, default: "" },
      rcBookImage:      { type: String, default: "" },         // Cloudinary URL
      insuranceExpiry:  { type: Date },
      insuranceImage:   { type: String, default: "" },         // Cloudinary URL
      vehicleImage:     { type: String, default: "" },         // Cloudinary URL (car photo)
    },

    // ── Approval ───────────────────────────────────────────────────────────
    isApproved:     { type: Boolean, default: false },
    approvedAt:     { type: Date },
    rejectionNote:  { type: String, default: "" },             // reason if admin rejects

    // ── Live Status ────────────────────────────────────────────────────────
    isOnline: { type: Boolean, default: false },

    // ── Current Location (GeoJSON — 2dsphere index for nearby queries) ─────
    currentLocation: {
      type:        { type: String, default: "Point" },
      coordinates: { type: [Number], default: [0, 0] },        // [longitude, latitude]
    },

    // ── Performance Stats ──────────────────────────────────────────────────
    stats: {
      totalRides:     { type: Number, default: 0 },
      completedRides: { type: Number, default: 0 },
      cancelledRides: { type: Number, default: 0 },
    },

    averageRating:        { type: Number, default: 5.0 },
    totalRatingsReceived: { type: Number, default: 0 },
    trustScore:           { type: Number, default: 100 },

    // ── Weekly Schedule (embedded) ─────────────────────────────────────────
    shifts: [
      {
        day:       { type: String, enum: ["Mon", "Tue", "Wed", "Thu", "Fri", "Sat", "Sun"] },
        startTime: { type: String },                           // "HH:MM" 24hr
        endTime:   { type: String },
        isActive:  { type: Boolean, default: true },
      },
    ],
  },
  { timestamps: true }
);

DriverProfileSchema.index({ currentLocation: "2dsphere" });
DriverProfileSchema.index({ isOnline: 1, isApproved: 1 });

module.exports = mongoose.model("DriverProfile", DriverProfileSchema);

3. TRIP  (covers searching phase + ride phase via "phase" field)
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

const TripSchema = new Schema(
  {
    passenger: { type: Schema.Types.ObjectId, ref: "User", required: true },
    driver:    { type: Schema.Types.ObjectId, ref: "User" },   // null until matched

    // ── Full lifecycle tracked by phase ────────────────────────────────────
    phase: {
      type: String,
      enum: [
        "searching",     // passenger requested, finding driver
        "matched",       // driver accepted, heading to pickup
        "ongoing",       // OTP verified, ride started
        "completed",     // ride finished
        "cancelled",     // cancelled at any phase
      ],
      default: "searching",
    },

    // ── Locations (GeoJSON) ────────────────────────────────────────────────
    source: {
      address:  { type: String, required: true },
      location: {
        type:        { type: String, default: "Point" },
        coordinates: { type: [Number], required: true },       // [lng, lat]
      },
    },

    destination: {
      address:  { type: String, required: true },
      location: {
        type:        { type: String, default: "Point" },
        coordinates: { type: [Number], required: true },
      },
    },

    vehicleTypeRequested: {
      type: String,
      enum: ["Sedan", "SUV", "Hatchback", "Auto", "Bike"],
    },

    // ── Driver Matching ────────────────────────────────────────────────────
    driversNotified: [{ type: Schema.Types.ObjectId, ref: "User" }],
    driversRejected: [{ type: Schema.Types.ObjectId, ref: "User" }],
    expiresAt:       { type: Date },                           // auto-cancel if no driver found

    // ── OTP ────────────────────────────────────────────────────────────────
    otp:         { type: String },                             // 4-digit, shown to passenger
    otpVerified: { type: Boolean, default: false },

    // ── Distance & Duration ────────────────────────────────────────────────
    distanceEstimate: { type: Number },                        // km  — shown before ride
    durationEstimate: { type: Number },                        // min — shown before ride
    distanceActual:   { type: Number },                        // km  — filled on completion
    durationActual:   { type: Number },                        // min — filled on completion

    // ── Fare ───────────────────────────────────────────────────────────────
    fare: {
      baseFare:     { type: Number, default: 0 },
      distanceFare: { type: Number, default: 0 },
      timeFare:     { type: Number, default: 0 },
      surgeFare:    { type: Number, default: 0 },
      discount:     { type: Number, default: 0 },
      total:        { type: Number, default: 0 },
    },

    surgeMultiplier: { type: Number, default: 1.0 },

    // ── Payment ────────────────────────────────────────────────────────────
    paymentMethod: { type: String, enum: ["wallet", "cash", "upi"] },
    paymentStatus: { type: String, enum: ["pending", "paid", "failed"], default: "pending" },

    // ── Cancellation ───────────────────────────────────────────────────────
    cancelledBy:        { type: String, enum: ["passenger", "driver", "system"] },
    cancellationReason: { type: String, default: "" },

    // ── Phase Timestamps ───────────────────────────────────────────────────
    matchedAt:       { type: Date },
    driverArrivedAt: { type: Date },
    startedAt:       { type: Date },
    completedAt:     { type: Date },
    cancelledAt:     { type: Date },
  },
  { timestamps: true }
);

TripSchema.index({ passenger: 1, phase: 1 });
TripSchema.index({ driver: 1, phase: 1 });
TripSchema.index({ createdAt: -1 });

module.exports = mongoose.model("Trip", TripSchema);

  TRUST SCORE LOGIC  (recalculate after every Review or Trip cancellation)
================================================================================

  Start:   100

  +1.0     for every 50 completed rides
  -1.0     for every trip cancelled by this user
  -5.0     for admin-flagged safety complaint
  Clamp:   min 0, max 100

  → Update passengerStats.trustScore in User  (for passengers)
  → Update trustScore in DriverProfile        (for drivers)
