1. USERS COLLECTION
--------------------------------------------------------------------------------

const UserSchema = new Schema({
  name:                 { type: String, required: true },
  email:                { type: String, required: true, unique: true },
  mobile_no:            { type: String, required: true, unique: true },
  password:             { type: String, required: true },           // bcrypt hashed
  role:                 { type: String, enum: ['passenger', 'driver', 'admin'], default: 'passenger' },
  profileImage:         { type: String },                           // URL
  isVerified:           { type: Boolean, default: false },          // [ADDED] phone/email verified?
  isBlocked:            { type: Boolean, default: false },          // [ADDED] admin can block user
  fcmToken:             { type: String },                           // [ADDED] for push notifications

  // [REMOVED] referralCode & referredBy — defer until referral system UI is built
  // [REMOVED] currentWalletBalance — single source of truth lives in Wallet collection
}, { timestamps: true });

Index: email (unique), mobile_no (unique)

3. DRIVER PROFILE

const DriverProfileSchema = new Schema({
  user:             { type: ObjectId, ref: 'User', required: true, unique: true },
  bio:              { type: String },

  license: {
    number:         { type: String },
    expiry:         { type: Date },
    frontImage:     { type: String },                               // URL
    backImage:      { type: String },                               // URL
  },

  aadhar: {
    number:         { type: String },
    image:          { type: String },                               // URL
  },

  vehicle: {
    name:           { type: String },                               // e.g. "Swift Dzire"
    type:           { type: String, enum: ['Sedan', 'SUV', 'Hatchback', 'Auto', 'Bike'] },
    number:         { type: String },                               // Registration number
    color:          { type: String },
    rcBookImage:    { type: String },                               // URL
    insuranceExpiry:{ type: Date },
    insuranceImage: { type: String },                               // [ADDED] URL
  },

  isApproved:       { type: Boolean, default: false },              // admin verification
  approvedAt:       { type: Date },                                 // [ADDED]

  // [MODIFIED] isOnline is OK for MVP in MongoDB; move to Redis in production
  isOnline:         { type: Boolean, default: false },

  currentLocation: {                                                // Geospatial — 2dsphere index
    type:           { type: String, default: 'Point' },
    coordinates:    { type: [Number] },                             // [longitude, latitude]
  },

  stats: {
    totalRides:       { type: Number, default: 0 },
    completedRides:   { type: Number, default: 0 },
    cancelledRides:   { type: Number, default: 0 },
    // [REMOVED] totalEarnings — derive from Wallet transactions to avoid sync bugs
  },

  averageRating:        { type: Number, default: 5.0 },
  totalRatingsReceived: { type: Number, default: 0 },
  trustScore:           { type: Number, default: 100 },

}, { timestamps: true });

Index: currentLocation (2dsphere)

4. RIDE REQUEST COLLECTION  [NEW — Pre-Match State]
--------------------------------------------------------------------------------
Purpose: Represents the "searching for a driver" phase BEFORE a Trip is created.
         A Trip is only created once a driver accepts.

const RideRequestSchema = new Schema({
  passenger:            { type: ObjectId, ref: 'User', required: true },

  source: {
    address:            { type: String },
    location: {
      type:             { type: String, default: 'Point' },
      coordinates:      { type: [Number] },                        // [lng, lat]
    }
  },

  destination: {
    address:            { type: String },
    location: {
      type:             { type: String, default: 'Point' },
      coordinates:      { type: [Number] },
    }
  },

  vehicleTypeRequested: { type: String, enum: ['Sedan', 'SUV', 'Hatchback', 'Auto', 'Bike'] },
  fareEstimate:         { type: Number },
  distanceEstimate:     { type: Number },                          // in km
  durationEstimate:     { type: Number },                          // in minutes

  status: {
    type: String,
    enum: ['searching', 'matched', 'expired', 'cancelled'],
    default: 'searching'
  },

  driversNotified:      [{ type: ObjectId, ref: 'User' }],        // drivers who were pinged
  driversRejected:      [{ type: ObjectId, ref: 'User' }],        // drivers who rejected
  matchedDriver:        { type: ObjectId, ref: 'User' },           // set once accepted
  tripId:               { type: ObjectId, ref: 'Trip' },           // set once Trip is created
  expiresAt:            { type: Date },                            // auto-cancel after N seconds

}, { timestamps: true });

Index: passenger, status, source.location (2dsphere)

5. TRIP / RIDE COLLECTION
--------------------------------------------------------------------------------

const TripSchema = new Schema({
  rideRequest:      { type: ObjectId, ref: 'RideRequest' },        // [ADDED] link to origin request
  passenger:        { type: ObjectId, ref: 'User', required: true },
  driver:           { type: ObjectId, ref: 'User', required: true },

  source: {
    address:        { type: String },
    location: {
      type:         { type: String, default: 'Point' },
      coordinates:  { type: [Number] },                            // [MODIFIED] GeoJSON format
    }
  },

  destination: {
    address:        { type: String },
    location: {
      type:         { type: String, default: 'Point' },
      coordinates:  { type: [Number] },
    }
  },

  otp:              { type: String },                              // [ADDED] 4-digit, passenger shows driver
  otpVerified:      { type: Boolean, default: false },            // [ADDED]

  distance:         { type: Number },                             // [ADDED] in km (actual travelled)
  duration:         { type: Number },                             // [ADDED] in minutes (actual)
  routePolyline:    { type: String },                             // [ADDED] encoded Google Maps path

  fare: {
    baseFare:       { type: Number },                             // [MODIFIED] broken down
    distanceFare:   { type: Number },
    timeFare:       { type: Number },
    surgeFare:      { type: Number, default: 0 },
    discount:       { type: Number, default: 0 },
    total:          { type: Number },
  },

  surgeMultiplier:  { type: Number, default: 1.0 },               // [ADDED]

  paymentMethod:    { type: String, enum: ['wallet', 'cash', 'upi'], required: true },  // [ADDED]
  paymentStatus:    { type: String, enum: ['pending', 'paid', 'failed'], default: 'pending' }, // [ADDED]

  status: {
    type: String,
    enum: ['accepted', 'driver_arriving', 'ongoing', 'completed', 'cancelled'],
    default: 'accepted'
  },

  cancelledBy:      { type: String, enum: ['passenger', 'driver', 'system'] },  // [ADDED]
  cancellationReason: { type: String },                           // [ADDED]

  requestedAt:      { type: Date, default: Date.now },
  driverArrivedAt:  { type: Date },                               // [ADDED]
  startedAt:        { type: Date },                               // [ADDED]
  completedAt:      { type: Date },                               // [ADDED]

}, { timestamps: true });

Index: passenger, driver, status, createdAt (desc)

  rating:       { type: Number, min: 1, max: 5, required: true },
  comment:      { type: String },
  tags:         [{ type: String }],
  // Predefined tag pool — enforce in application layer:
  // FOR DRIVER:     ['Polite', 'Clean Car', 'On Time', 'Safe Driving', 'Good Route']
  // FOR PASSENGER:  ['Good Behavior', 'Ready on Time', 'Respectful', 'No Mess']
}, { timestamps: true });

// [ADDED] Compound unique index — one review per reviewer per trip
Index: { reviewer: 1, trip: 1 } (unique)


--------------------------------------------------------------------------------
TRUST SCORE ALGORITHM
--------------------------------------------------------------------------------
Base Score: 100
(+) +0.5 for every 5-star review
(+) +1.0 for every 50 completed rides
(-) -1.0 for every cancelled trip (if at fault)
(-) -5.0 for safety/complaint flag by admin
Recalculate: on every new Review saved OR Trip cancelled via background worker.
Cap: Min 0, Max 100

10. SOS / EMERGENCY  [NEW]
--------------------------------------------------------------------------------
Purpose: Passenger triggers SOS during an active ride. Critical safety feature.

const SOSSchema = new Schema({
  trip:           { type: ObjectId, ref: 'Trip', required: true },
  passenger:      { type: ObjectId, ref: 'User', required: true },
  location: {
    type:         { type: String, default: 'Point' },
    coordinates:  { type: [Number] },
  },
  status:         { type: String, enum: ['active', 'resolved'], default: 'active' },
  triggeredAt:    { type: Date, default: Date.now },
  resolvedAt:     { type: Date },
  resolvedBy:     { type: ObjectId, ref: 'User' },               // admin who resolved
  notes:          { type: String },
}, { timestamps: true });
